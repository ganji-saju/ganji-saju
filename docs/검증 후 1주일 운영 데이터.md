SELECT feature,\
  count(*) FILTER (WHERE source='cache') as hits,\
  count(*) FILTER (WHERE source='openai') as misses,\
  count(*) FILTER (WHERE source='fallback') as fallbacks,\
  round(100.0 * count(*) FILTER (WHERE source='cache') / count(*), 1) as hit_rate_pct\

WHERE created_at > now() - interval '1 day'\
GROUP BY feature;\
